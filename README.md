create database hospital;
use hospital;
CREATE TABLE signup (
    patientNu INT NOT NULL AUTO_INCREMENT,
    email VARCHAR(255) NOT NULL,
    Name VARCHAR(255) NOT NULL,
    Age INT NOT NULL,
    ContactNumber VARCHAR(15) NOT NULL,
    password VARCHAR(255) NOT NULL,
    service VARCHAR(255),
    doctor VARCHAR(255),
    PRIMARY KEY (patientNu)
);
